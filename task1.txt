#include <iostream>  // Підключення бібліотеки iostream для введення та виведення (std::cout і std::cin)
#include <cmath>     // Підключення бібліотеки cmath для використання математичних функцій, таких як exp(), sin() і log()

int main()           // Точка входу в програму
{
    int a;           // Оголошення змінної типу int 'a' для зберігання значення 'a'
    int b;           // Оголошення змінної типу int 'b' для зберігання значення 'b'

    std::cout << "Введіть а: ";  // Виведення повідомлення з проханням ввести значення для 'a'
    std::cin >> a;               // Зчитування введеного користувачем значення та збереження його в змінну 'a'

    std::cout << "Введіть b: ";  // Виведення повідомлення з проханням ввести значення для 'b'
    std::cin >> b;               // Зчитування введеного користувачем значення та збереження його в змінну 'b'

    // Обчислення чисельника за формулою: (0.314 * e^a) - (0.512 * e^b)
    // exp(a) обчислює e^a (експоненціальна функція), аналогічно для exp(b)
    double numerator = 0.314 * exp(a) - 0.512 * exp(b);

    // Обчислення знаменника за формулою: sin((b/3) * pi) * ln(a)
    // M_PI — це константа, що представляє значення числа пі (π), визначена в бібліотеці cmath
    // log(a) обчислює натуральний логарифм (ln) від 'a'
    double denominator = sin((b / 3) * M_PI) * log(a);  // M_PI — це значення пі з бібліотеки cmath

    // Перевірка на нульовий знаменник, щоб уникнути ділення на нуль
    if (denominator == 0) {
        std::cout << "Помилка: Ділення на нуль!" << std::endl;  // Виведення повідомлення про помилку
    } else {
        // Обчислення результату шляхом ділення чисельника на знаменник
        double result = numerator / denominator;
        std::cout << "Результат: " << result << std::endl;  // Виведення результату
    }

    return 0;  // Завершення програми, повернення 0, що означає успішне виконання
}
